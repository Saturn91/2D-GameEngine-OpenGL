#version 400 core

in vec3 position;
in vec2 textureCoords;

out vec2 pass_textureCoords;	//same name as the in in fragmentShader!!
out vec3 pass_position;
out mat4 pass_view_projection;

uniform mat4 transformationMatrix;
uniform mat4 projectionMatrix;
uniform mat4 viewMatrix;

void main(void){
	pass_view_projection = projectionMatrix * viewMatrix;
	gl_Position = pass_view_projection * transformationMatrix * vec4(position.x, position.y, position.z, 1.0);
	pass_textureCoords = textureCoords;
	pass_position = vec3(pass_view_projection * transformationMatrix * vec4(position.x, position.y, position.z, 1.0));

}
